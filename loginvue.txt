<template>
  <div class="login-container">
    <div class="login-background">

      <div class="floating-shape shape-2"></div>
      <div class="floating-shape shape-3"></div>
    </div>
    
    <div class="login-card">
      <div class="login-header">
        <div class="logo-container">
          <div class="logo-icon">üõ°Ô∏è</div>
          <h1 class="login-title">Welcome Back</h1>
          <p class="login-subtitle">Sign in to your abuse moderation dashboard</p>
        </div>
      </div>
      
      <form @submit.prevent="handleLogin" class="login-form">
        <div class="form-group">
          <div class="input-container">
            <input
              v-model="username"
              type="text"
              placeholder="Enter your username"
              class="form-input"
              required
            />
            <div class="input-icon">üë§</div>
          </div>
        </div>
        
        <div class="form-group">

              v-model="password"

            />

          </div>
        </div>
        
        <button type="submit" class="login-button" :disabled="loading">
          <span v-if="loading" class="loading-spinner"></span>
          <span v-else>{{ loading ? 'Signing in...' : 'Sign In' }}</span>
        </button>
        
        <div v-if="error" class="error-message">
          <div class="error-icon">‚ö†Ô∏è</div>
          {{ error }}
        </div>

        <div class="divider">
          <span>New to the platform?</span>

        <router-link to="/register" class="register-link">
          Create your account
        </router-link>
      </form>
    </div>
  </div>
</template>

<script>
import { authAPI } from '@/services/api'

export default {
  name: 'Login',
  data() {
    return {
      username: '',
      password: '',
      loading: false,
      error: ''
    }
  },
  mounted() {
    // Add entrance animation
    setTimeout(() => {
      document.querySelector('.login-card').classList.add('animate-in')
    }, 100)
  },
  methods: {
    async handleLogin() {
      this.loading = true
      this.error = ''
      
      try {
        const response = await authAPI.login({
          username: this.username,
          password: this.password
        })
        
        const { access_token, user } = response
        
        localStorage.setItem('token', access_token)
        localStorage.setItem('user', JSON.stringify(user))
        
        if (user.role === 'moderator') {
          this.$router.push('/mod-dashboard')
        } else {
          this.$router.push('/feed')
        }
        
      } catch (error) {
        this.error = error.response?.data?.detail || 'Login failed. Please try again.'
        console.error('Login error:', error)
      } finally {
        this.loading = false
      }
    }
  }
}
</script>
